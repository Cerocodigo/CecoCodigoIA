


modulos: son los procesos y objetos del negocio. 
Ejemplo:
[
    {
        "_id": "clientes", // Identificador único del módulo (clave interna, no debe repetirse)
        "nombre": "Clientes", // Nombre visible del módulo en la interfaz
        "descripcion": "Lista de clientes de empresa", // Explicación funcional de lo que hace el módulo 
        "uso": "alto", // Nivel de uso esperado (puede servir para priorizar UI, rendimiento o análisis)
        "activo": true, // Indica si el módulo está habilitado o disponible en el sistema
        "entidades": [
            "Clientes" // Lista de entidades/tablas/modelos que pertenecen a este módulo
        ], // Listado de las tablas vinculadas al modulo
        "creado_en": {
            "$date": "2026-02-11T19:41:39.998Z" // Fecha de creación del módulo en formato ISO (útil para auditoría)
        },
        "prompt_config":"Proposito":"Mantener el listado de clientes, para ser usado por otros modulos"
    }
]
